package data

import com.google.gson.annotations.Expose
import com.google.gson.annotations.SerializedName


data class Customer(
    var latitude: Double?,
    var longitude: Double?, @Expose @SerializedName("user_id") var userId: Int?, @Expose val name: String?
) {

    fun sphereLocation(): SphereLocation {
        return SphereLocation(latitude ?: 0.0, longitude ?: 0.0)
    }
}
